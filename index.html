const chatList = document.getElementById("chatList"); const deleteCurrentChatButton = document.getElementById("deleteCurrentChatButton"); const deleteAllChatsButton = document.getElementById("deleteAllChatsButton"); function deleteChat(chatId) { const chats = JSON.parse(localStorage.getItem("talk2me.chats") || "[]"); const nextChats = chats.filter(chat => chat.id !== chatId); localStorage.setItem("talk2me.chats", JSON.stringify(nextChats)); const activeChatId = localStorage.getItem("talk2me.activeChatId"); if (activeChatId === chatId) { localStorage.removeItem("talk2me.activeChatId"); } renderChatList(); renderActiveChat(); } function deleteCurrentChat() { const activeChatId = localStorage.getItem("talk2me.activeChatId"); if (!activeChatId) return; const confirmed = confirm("Delete this chat?"); if (!confirmed) return; deleteChat(activeChatId); } function deleteAllChats() { const confirmed = confirm("Delete all chats? This cannot be undone."); if (!confirmed) return; localStorage.removeItem("talk2me.chats"); localStorage.removeItem("talk2me.activeChatId"); renderChatList(); renderActiveChat(); } chatList?.addEventListener("click", event => { const deleteButton = event.target.closest(".delete-chat-button"); if (!deleteButton) return; const chatItem = deleteButton.closest("[data-chat-id]"); if (!chatItem) return; const confirmed = confirm("Delete this chat?"); if (!confirmed) return; deleteChat(chatItem.dataset.chatId); }); deleteCurrentChatButton?.addEventListener("click", deleteCurrentChat); deleteAllChatsButton?.addEventListener("click", deleteAllChats);
